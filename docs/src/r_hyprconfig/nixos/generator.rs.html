52    /// Generate NixOS configuration from current system state
53    pub async fn generate_from_system(
54        &self,
55        options: &NixGenerationOptions,
56    ) -> Result<GeneratedNixConfig> {
57        // Detect if we're on NixOS
58        let distribution = DistributionDetector::detect()?;
59        if distribution.distribution_type != DistributionType::NixOS {
60            anyhow::bail!(
61                "NixOS configuration generation is only supported on NixOS systems. Detected: {}",
62                distribution.distribution_type
63            );
64        }
65
66        // Get current hyprland configuration
67        let hyprctl = crate::hyprctl::HyprCtl::new().await?;
68        let current_config = self.extract_current_config(&hyprctl).await?;
69
70        self.generate_config(&current_config, options)
71    }

82    /// Generate configuration with advanced template support
83    fn generate_config(
84        &self,
85        config: &HyprlandConfig,
86        options: &NixGenerationOptions,
87    ) -> Result<GeneratedNixConfig> {
88        let settings = self.extract_settings(config);
89        let keybinds = self.extract_keybinds(config);
90        let window_rules = config.window_rules.clone();
91        let layer_rules = config.layer_rules.clone();
92
93        let nix_content = match &options.config_type {
94            NixConfigType::HomeManager => {
95                self.generate_enhanced_home_manager(&settings, &keybinds, &window_rules, &layer_rules, options)?
96            }
97            NixConfigType::SystemConfig => {
98                self.generate_enhanced_system_config(&settings, &keybinds, &window_rules, &layer_rules, options)?
99            }
100            NixConfigType::FlakeHomeManager => {
101                self.generate_enhanced_flake_home_manager(&settings, &keybinds, &window_rules, &layer_rules, options)?
102            }
103            NixConfigType::FlakeSystem => {
104                self.generate_enhanced_flake_system(&settings, &keybinds, &window_rules, &layer_rules, options)?
105            }
106        };
107
108        let output_path = self.determine_output_path(options)?;
109
110        Ok(GeneratedNixConfig {
111            content: nix_content,
112            config_type: options.config_type.clone(),
113            output_path,
114            module_name: options.module_name.clone(),
115            metadata: GenerationMetadata {
116                generated_at: chrono::Utc::now(),
117                source_distribution: DistributionDetector::detect()?.distribution_type,
118                generator_version: env!("CARGO_PKG_VERSION").to_string(),
119                options: options.clone(),
120            },
121        })
122    }

124    /// Generate enhanced Home Manager configuration
125    fn generate_enhanced_home_manager(
126        &self,
127        settings: &HashMap<String, String>,
128        keybinds: &[String],
129        window_rules: &[String],
130        layer_rules: &[String],
131        options: &NixGenerationOptions,
132    ) -> Result<String> {
133        let mut nix_config = String::new();
134
135        // File header with metadata
136        nix_config.push_str(&self.generate_header(&options.module_name)?);
137
138        // Imports section
139        nix_config.push_str("{ config, lib, pkgs, ... }:\n\n");
140        nix_config.push_str("{\n");
141
142        if !options.custom_imports.is_empty() {
143            nix_config.push_str("  imports = [\n");
144            for import in &options.custom_imports {
145                nix_config.push_str(&format!("    {}\n", import));
146            }
147            nix_config.push_str("  ];\n\n");
148        }
149
150        // Wayland environment
151        if options.enable_wayland {
152            nix_config.push_str(&self.generate_wayland_config()?);
153        }
154
155        // Programs section
156        if options.include_programs {
157            nix_config.push_str(&self.generate_programs_config(options)?);
158        }
159
160        // Main Hyprland configuration
161        nix_config.push_str("  wayland.windowManager.hyprland = {\n");
162        nix_config.push_str("    enable = true;\n");
163        
164        if options.enable_nvidia {
165            nix_config.push_str("    enableNvidiaPatches = true;\n");
166        }
167
168        nix_config.push_str("    settings = {\n");
169        
170        // General settings
171        nix_config.push_str(&self.generate_nix_settings(settings)?);
172        
173        // Keybinds
174        if !keybinds.is_empty() {
175            nix_config.push_str("      bind = [\n");
176            for keybind in keybinds {
177                nix_config.push_str(&format!("        \"{}\"\n", self.escape_nix_string(keybind)));
178            }
179            nix_config.push_str("      ];\n");
180        }
181
182        // Window rules
183        if !window_rules.is_empty() {
184            nix_config.push_str("      windowrule = [\n");
185            for rule in window_rules {
186                nix_config.push_str(&format!("        \"{}\"\n", self.escape_nix_string(rule)));
187            }
188            nix_config.push_str("      ];\n");
189        }
190
191        // Layer rules
192        if !layer_rules.is_empty() {
193            nix_config.push_str("      layerrule = [\n");
194            for rule in layer_rules {
195                nix_config.push_str(&format!("        \"{}\"\n", self.escape_nix_string(rule)));
196            }
197            nix_config.push_str("      ];\n");
198        }
199
200        nix_config.push_str("    };\n");
201        nix_config.push_str("  };\n");
202
203        // Systemd services
204        if options.include_systemd_services {
205            nix_config.push_str(&self.generate_systemd_services()?);
206        }
207
208        nix_config.push_str("}\n");
209
210        Ok(nix_config)
211    }

213    /// Generate enhanced system configuration
214    fn generate_enhanced_system_config(
215        &self,
216        _settings: &HashMap<String, String>,
217        _keybinds: &[String],
218        _window_rules: &[String],
219        _layer_rules: &[String],
220        options: &NixGenerationOptions,
221    ) -> Result<String> {
222        let mut nix_config = String::new();
223
224        nix_config.push_str(&self.generate_header(&options.module_name)?);
225        nix_config.push_str("{ config, lib, pkgs, ... }:\n\n");
226        nix_config.push_str("{\n");
227
228        // System-level Hyprland configuration
229        nix_config.push_str("  programs.hyprland = {\n");
230        nix_config.push_str("    enable = true;\n");
231        
232        if options.enable_nvidia {
233            nix_config.push_str("    enableNvidiaPatches = true;\n");
234        }
235        
236        nix_config.push_str("  };\n\n");
237
238        // Environment variables
239        nix_config.push_str("  environment.sessionVariables = {\n");
240        if options.enable_wayland {
241            nix_config.push_str("    NIXOS_OZONE_WL = \"1\";\n");
242            nix_config.push_str("    MOZ_ENABLE_WAYLAND = \"1\";\n");
243        }
244        if options.enable_nvidia {
245            nix_config.push_str("    LIBVA_DRIVER_NAME = \"nvidia\";\n");
246            nix_config.push_str("    GBM_BACKEND = \"nvidia-drm\";\n");
247            nix_config.push_str("    __GLX_VENDOR_LIBRARY_NAME = \"nvidia\";\n");
248        }
249        nix_config.push_str("  };\n\n");
250
251        // Hardware configuration
252        if options.enable_nvidia {
253            nix_config.push_str(&self.generate_nvidia_config()?);
254        }
255
256        // Default configuration hint
257        nix_config.push_str("  # Note: User-specific Hyprland configuration should be managed through Home Manager\n");
258        nix_config.push_str("  # This system configuration only enables Hyprland system-wide\n");
259
260        nix_config.push_str("}\n");
261
262        Ok(nix_config)
263    }

276        nix_config.push_str(&self.generate_header(&options.module_name)?);

285        nix_config.push_str("  };\n\n");
286
287        nix_config.push_str("  outputs = { nixpkgs, home-manager, hyprland, ... }:\n");
288        nix_config.push_str("    let\n");
289        nix_config.push_str("      system = \"x86_64-linux\";\n");
290        nix_config.push_str("      pkgs = nixpkgs.legacyPackages.${system};\n");
291        nix_config.push_str("    in {\n");
292        nix_config.push_str("      homeConfigurations.\"$USER\" = home-manager.lib.homeManagerConfiguration {\n");
293        nix_config.push_str("        inherit pkgs;\n");
294        nix_config.push_str("        modules = [\n");
295        nix_config.push_str("          hyprland.homeManagerModules.default\n");
296        nix_config.push_str("          {\n");
297
298        // Include the home manager configuration inline
299        let home_config = self.generate_enhanced_home_manager(settings, keybinds, window_rules, layer_rules, options)?;
300        // Remove the outer wrapper and add proper indentation
301        let lines: Vec<&str> = home_config.lines().collect();
302        for line in lines.iter().skip(3).take(lines.len() - 4) { // Skip header and wrapper
303            nix_config.push_str("            ");
304            nix_config.push_str(line);
305            nix_config.push('\n');
306        }
307
308        nix_config.push_str("          }\n");
309        nix_config.push_str("        ];\n");
310        nix_config.push_str("      };\n");
311        nix_config.push_str("    };\n");
312        nix_config.push_str("}\n");
313
314        Ok(nix_config)
315    }

402    fn generate_nix_settings(&self, settings: &HashMap<String, String>) -> Result<String> {
403        let mut nix_settings = String::new();
404
405        // Group settings by category
406        let categories = [
407            ("general", "General settings"),
408            ("input", "Input configuration"), 
409            ("decoration", "Visual decoration"),
410            ("animations", "Animation settings"),
411            ("gestures", "Gesture configuration"),
412            ("misc", "Miscellaneous settings"),
413        ];
414
415        for (category, description) in &categories {
416            let category_settings: HashMap<String, String> = settings
417                .iter()
418                .filter(|(key, _)| key.starts_with(&format!("{}:", category)))
419                .map(|(key, value)| (key.strip_prefix(&format!("{}:", category)).unwrap().to_string(), value.clone()))
420                .collect();
421
422            if !category_settings.is_empty() {
423                nix_settings.push_str(&format!("      # {}\n", description));
424                nix_settings.push_str(&format!("      {} = {{\n", category));
425                
426                for (key, value) in &category_settings {
427                    let nix_value = self.convert_value_to_nix(value)?;
428                    nix_settings.push_str(&format!("        {} = {};\n", key, nix_value));
429                }
430                
431                nix_settings.push_str("      };\n");
432            }
433        }
434
435        Ok(nix_settings)
436    }

476    fn convert_value_to_nix(&self, value: &str) -> Result<String> {
477        // Handle different value types for Nix
478        if value == "true" || value == "false" {
479            Ok(value.to_string())
480        } else if value.parse::<i32>().is_ok() {
481            Ok(value.to_string())
482        } else if value.parse::<f32>().is_ok() {
483            Ok(value.to_string())
484        } else if value.starts_with("rgba(") || value.starts_with("rgb(") {
485            Ok(format!("\"{}\"", value))
486        } else {
487            // String value
488            Ok(format!("\"{}\"", self.escape_nix_string(value)))
489        }
490    }
491
492    fn escape_nix_string(&self, value: &str) -> String {
493        value
494            .replace('\\', "\\\\")
495            .replace('"', "\\\"")
496            .replace('\n', "\\n")
497            .replace('\t', "\\t")
498    }

500    fn determine_output_path(&self, options: &NixGenerationOptions) -> Result<PathBuf> {
501        if let Some(ref path) = options.output_path {
502            return Ok(path.clone());
503        }
504
505        let paths = ConfigPathManager::get_paths()?;
506        let filename = match &options.config_type {
507            NixConfigType::HomeManager => format!("{}-home.nix", options.module_name),
508            NixConfigType::SystemConfig => format!("{}-system.nix", options.module_name),
509            NixConfigType::FlakeHomeManager => "flake-home.nix".to_string(),
510            NixConfigType::FlakeSystem => "flake.nix".to_string(),
511        };
512
513        Ok(paths.exports_dir.join(filename))
514    }

516    async fn extract_current_config(&self, hyprctl: &crate::hyprctl::HyprCtl) -> Result<HyprlandConfig> {
517        let options = hyprctl.get_all_options().await?;
518        let binds = hyprctl.get_binds().await?;
519        let window_rules = hyprctl.get_window_rules().await?;
520        let layer_rules = hyprctl.get_layer_rules().await?;
521
522        // Convert options to structured config
523        Ok(HyprlandConfig {
524            general: crate::hyprctl::GeneralConfig {
525                gaps_in: options.get("general:gaps_in").and_then(|v| v.parse().ok()).unwrap_or(5),
526                gaps_out: options.get("general:gaps_out").and_then(|v| v.parse().ok()).unwrap_or(20),
527                border_size: options.get("general:border_size").and_then(|v| v.parse().ok()).unwrap_or(2),
528                col_active_border: options.get("general:col.active_border").cloned().unwrap_or_else(|| "rgba(33ccffee)".to_string()),
529                col_inactive_border: options.get("general:col.inactive_border").cloned().unwrap_or_else(|| "rgba(595959aa)".to_string()),
530                resize_on_border: options.get("general:resize_on_border").map(|v| v == "true").unwrap_or(false),
531                extend_border_grab_area: options.get("general:extend_border_grab_area").and_then(|v| v.parse().ok()).unwrap_or(15),
532                hover_icon_on_border: options.get("general:hover_icon_on_border").map(|v| v == "true").unwrap_or(true),
533            },
534            input: Default::default(),
535            decoration: Default::default(),
536            animations: Default::default(),
537            gestures: Default::default(),
538            binds: binds.iter().map(|b| b.display_string()).collect(),
539            window_rules,
540            layer_rules,
541            misc: Default::default(),
542        })
543    }

565/// Result of NixOS configuration generation
566#[derive(Debug, Clone)]
567pub struct GeneratedNixConfig {
568    pub content: String,
569    pub config_type: NixConfigType,
570    pub output_path: PathBuf,
571    pub module_name: String,
572    pub metadata: GenerationMetadata,
573}
574
575#[derive(Debug, Clone, Serialize, Deserialize)]
576pub struct GenerationMetadata {
577    pub generated_at: chrono::DateTime<chrono::Utc>,
578    pub source_distribution: DistributionType,
579    pub generator_version: String,
580    pub options: NixGenerationOptions,
581}
582
583impl GeneratedNixConfig {
584    /// Save the generated configuration to disk
585    pub fn save(&self) -> Result<()> {
586        // Ensure directory exists
587        if let Some(parent) = self.output_path.parent() {
588            std::fs::create_dir_all(parent)
589                .with_context(|| format!("Failed to create directory: {:?}", parent))?;
590        }
591
592        std::fs::write(&self.output_path, &self.content)
593            .with_context(|| format!("Failed to write config to: {:?}", self.output_path))?;
594
595        Ok(())
596    }
597
598    /// Get a preview of the generated content (first 20 lines)
599    pub fn preview(&self) -> String {
600        self.content
601            .lines()
602            .take(20)
603            .collect::<Vec<_>>()
604            .join("\n")
605    }
