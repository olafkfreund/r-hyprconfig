14    // Convert traditional hyprland.conf format to NixOS format
15    pub fn traditional_to_nixos(
16        &self,
17        traditional_config: &HashMap<String, String>,
18        keybinds: &[String],
19        window_rules: &[String],
20        layer_rules: &[String],
21        target_type: NixConfigType,
22    ) -> Result<String, anyhow::Error> {
23        match target_type {
24            NixConfigType::HomeManager => self.generate_home_manager_config(
25                traditional_config,
26                keybinds,
27                window_rules,
28                layer_rules,
29            ),
30            NixConfigType::SystemConfig => {
31                self.generate_system_config(traditional_config, keybinds, window_rules, layer_rules)
32            }
33            NixConfigType::FlakeHomeManager => self.generate_flake_home_manager_config(
34                traditional_config,
35                keybinds,
36                window_rules,
37                layer_rules,
38            ),
39            NixConfigType::FlakeSystem => self.generate_flake_system_config(
40                traditional_config,
41                keybinds,
42                window_rules,
43                layer_rules,
44            ),
45        }
46    }

48    // Convert NixOS config back to traditional format
49    #[allow(dead_code, clippy::type_complexity)]
50    pub fn nixos_to_traditional(
51        &self,
52        nixos_config: &NixOSConfig,
53    ) -> Result<
54        (
55            HashMap<String, String>,
56            Vec<String>,
57            Vec<String>,
58            Vec<String>,
59        ),
60        anyhow::Error,
61    > {
62        let mut traditional_config = HashMap::new();
63        let mut keybinds = Vec::new();
64        let mut window_rules = Vec::new();
65        let mut layer_rules = Vec::new();
66
67        // Convert NixOS settings back to traditional format
68        for (key, value) in &nixos_config.hyprland_settings {
69            traditional_config.insert(key.clone(), value.clone());
70        }
71
72        // Convert keybinds
73        for keybind in &nixos_config.keybinds {
74            keybinds.push(self.nix_keybind_to_traditional(keybind));
75        }
76
77        // Convert window rules
78        for rule in &nixos_config.window_rules {
79            window_rules.push(self.nix_rule_to_traditional(rule, "windowrule"));
80        }
81
82        // Convert layer rules
83        for rule in &nixos_config.layer_rules {
84            layer_rules.push(self.nix_rule_to_traditional(rule, "layerrule"));
85        }
86
87        Ok((traditional_config, keybinds, window_rules, layer_rules))
88    }

90    // Generate Nix configuration template
91    #[allow(dead_code)]
92    pub fn generate_template(&self, config_type: NixConfigType) -> String {
93        match config_type {
94            NixConfigType::HomeManager => self.generate_home_manager_template(),
95            NixConfigType::SystemConfig => self.generate_system_config_template(),
96            NixConfigType::FlakeHomeManager => self.generate_flake_home_manager_template(),
97            NixConfigType::FlakeSystem => self.generate_flake_system_template(),
98        }
99    }

101    fn generate_home_manager_config(
102        &self,
103        traditional_config: &HashMap<String, String>,
104        keybinds: &[String],
105        window_rules: &[String],
106        layer_rules: &[String],
107    ) -> Result<String, anyhow::Error> {
108        let mut nix_config = String::new();
109
110        nix_config.push_str("{ config, pkgs, ... }: {\n");
111        nix_config.push_str("  wayland.windowManager.hyprland = {\n");
112        nix_config.push_str("    enable = true;\n");
113        nix_config.push_str("    settings = {\n");
114
115        // Convert traditional config sections to Nix attribute sets
116        let config_sections = self.group_config_by_section(traditional_config);
117
118        for (section, options) in config_sections {
119            if !options.is_empty() {
120                nix_config.push_str(&format!("      {section} = {{\n"));
121                for (key, value) in options {
122                    let nix_value = self.convert_value_to_nix(&value);
123                    nix_config.push_str(&format!("        {key} = {nix_value};\n"));
124                }
125                nix_config.push_str("      };\n");
126            }
127        }
128
129        // Add keybinds
130        if !keybinds.is_empty() {
131            nix_config.push_str("      bind = [\n");
132            for keybind in keybinds {
133                let clean_bind = self.convert_keybind_to_nix(keybind);
134                nix_config.push_str(&format!("        \"{clean_bind}\"\n"));
135            }
136            nix_config.push_str("      ];\n");
137        }
138
139        // Add window rules
140        if !window_rules.is_empty() {
141            nix_config.push_str("      windowrule = [\n");
142            for rule in window_rules {
143                let clean_rule = self.convert_rule_to_nix(rule);
144                nix_config.push_str(&format!("        \"{clean_rule}\"\n"));
145            }
146            nix_config.push_str("      ];\n");
147        }
148
149        // Add layer rules
150        if !layer_rules.is_empty() {
151            nix_config.push_str("      layerrule = [\n");
152            for rule in layer_rules {
153                let clean_rule = self.convert_rule_to_nix(rule);
154                nix_config.push_str(&format!("        \"{clean_rule}\"\n"));
155            }
156            nix_config.push_str("      ];\n");
157        }
158
159        nix_config.push_str("    };\n");
160        nix_config.push_str("  };\n");
161        nix_config.push_str("}\n");
162
163        Ok(nix_config)
164    }

262    // Helper methods for conversion
263    fn group_config_by_section(
264        &self,
265        config: &HashMap<String, String>,
266    ) -> HashMap<String, Vec<(String, String)>> {
267        let mut sections: HashMap<String, Vec<(String, String)>> = HashMap::new();
268
269        for (key, value) in config {
270            let (section, option) = if key.contains(':') {
271                let parts: Vec<&str> = key.splitn(2, ':').collect();
272                (parts[0].to_string(), parts[1].to_string())
273            } else {
274                // If no section prefix, assume it's general
275                ("general".to_string(), key.clone())
276            };
277
278            sections
279                .entry(section)
280                .or_default()
281                .push((option, value.clone()));
282        }
283
284        sections
285    }

287    fn convert_value_to_nix(&self, value: &str) -> String {
288        // Handle different value types for Nix
289        if value == "true" || value == "false" {
290            // Boolean values don't need quotes
291            value.to_string()
292        } else if value.parse::<i32>().is_ok() || value.parse::<f32>().is_ok() {
293            // Numeric values don't need quotes
294            value.to_string()
295        } else if value.starts_with("rgb(") || value.starts_with("rgba(") {
296            // Color values - keep as string but format nicely
297            format!("\"{value}\"")
298        } else {
299            // String values need quotes
300            format!("\"{}\"", self.escape_nix_string(value))
301        }
302    }

304    fn convert_keybind_to_nix(&self, keybind: &str) -> String {
305        // Convert from "bind = SUPER, Q, exec, kitty" to "SUPER, Q, exec, kitty"
306        if keybind.starts_with("bind = ") {
307            keybind
308                .strip_prefix("bind = ")
309                .unwrap_or(keybind)
310                .to_string()
311        } else if keybind.contains(" → ") {
312            // Convert from display format "SUPER + Q → exec [kitty]" to config format
313            self.display_keybind_to_config(keybind)
314        } else {
315            keybind.to_string()
316        }
317    }
318
319    fn convert_rule_to_nix(&self, rule: &str) -> String {
320        // Convert from "windowrule = float, ^(kitty)$" to "float, ^(kitty)$"
321        if rule.starts_with("windowrule = ") {
322            rule.strip_prefix("windowrule = ")
323                .unwrap_or(rule)
324                .to_string()
325        } else if rule.starts_with("layerrule = ") {
326            rule.strip_prefix("layerrule = ")
327                .unwrap_or(rule)
328                .to_string()
329        } else {
330            rule.to_string()
331        }
332    }

334    fn display_keybind_to_config(&self, display_keybind: &str) -> String {
335        // Convert "SUPER + Q → exec [kitty]" to "SUPER, Q, exec, kitty"
336        let parts: Vec<&str> = display_keybind.split(" → ").collect();
337        if parts.len() == 2 {
338            let modifiers_and_key = parts[0].replace(" + ", ", ");
339            let action_part = parts[1];
340
341            // Remove brackets from individual arguments like "exec [kitty]" -> "exec kitty"
342            let cleaned_action = action_part.replace(['[', ']'], "");
343            let action_and_args = cleaned_action.replace(' ', ", ");
344            format!("{modifiers_and_key}, {action_and_args}")
345        } else {
346            display_keybind.to_string()
347        }
348    }

350    fn escape_nix_string(&self, s: &str) -> String {
351        // Escape special characters for Nix strings
352        s.replace('\\', "\\\\")
353            .replace('"', "\\\"")
354            .replace('\n', "\\n")
355            .replace('\t', "\\t")
356    }

359    fn generate_system_config(
360        &self,
361        _traditional_config: &HashMap<String, String>,
362        _keybinds: &[String],
363        _window_rules: &[String],
364        _layer_rules: &[String],

370    # User configuration should be in Home Manager
371  };

376    fn generate_flake_home_manager_config(
377        &self,
378        traditional_config: &HashMap<String, String>,
379        keybinds: &[String],
380        window_rules: &[String],
381        layer_rules: &[String],
382    ) -> Result<String, anyhow::Error> {
383        let home_manager_config = self.generate_home_manager_config(
384            traditional_config,
385            keybinds,
386            window_rules,
387            layer_rules,
388        )?;
389
390        // Wrap in flake structure
391        let flake_config = format!(
392            r#"{{
393  description = "Home Manager flake with Hyprland configuration";
394  
395  inputs = {{
396    nixpkgs.url = "github:NixOS/nixpkgs/nixos-unstable";
397    home-manager = {{
398      url = "github:nix-community/home-manager";
399      inputs.nixpkgs.follows = "nixpkgs";
400    }};
401    hyprland.url = "github:hyprwm/Hyprland";
402  }};
403  
404  outputs = {{ nixpkgs, home-manager, hyprland, ... }}: {{
405    homeConfigurations."${{USER}}" = home-manager.lib.homeManagerConfiguration {{
406      pkgs = nixpkgs.legacyPackages.x86_64-linux;
407      modules = [
408        hyprland.homeManagerModules.default
409        ({home_manager_config})
410      ];
411    }};
412  }};
413}}"#
414        );
415
416        Ok(flake_config)
417    }

468    fn test_traditional_to_nixos_conversion() {
469        let converter = ConfigConverter::new();
470
471        // Test data
472        let mut traditional_config = HashMap::new();
473        traditional_config.insert("general:gaps_in".to_string(), "5".to_string());
474        traditional_config.insert("general:gaps_out".to_string(), "10".to_string());
475        traditional_config.insert("decoration:rounding".to_string(), "8".to_string());
476
477        let keybinds = vec![
478            "SUPER, Q, exec, kitty".to_string(),

481
482        let window_rules = vec!["float, ^(kitty)$".to_string()];
483
484        let layer_rules = vec!["blur, waybar".to_string()];
485
486        // Convert to NixOS Home Manager format
487        let result = converter.traditional_to_nixos(
488            &traditional_config,
489            &keybinds,
490            &window_rules,
491            &layer_rules,
492            NixConfigType::HomeManager,
493        );
494
495        assert!(result.is_ok());
496        let nix_config = result.unwrap();
497
498        // Verify the structure
499        assert!(nix_config.contains("wayland.windowManager.hyprland"));
500        assert!(nix_config.contains("enable = true"));
501        assert!(nix_config.contains("settings = {"));
502        assert!(nix_config.contains("general = {"));
503        assert!(nix_config.contains("gaps_in = 5"));
504        assert!(nix_config.contains("gaps_out = 10"));
505        assert!(nix_config.contains("decoration = {"));
506        assert!(nix_config.contains("rounding = 8"));
