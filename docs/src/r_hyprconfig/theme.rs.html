318    pub fn data_type_style(&self, data_type: &crate::ui::ConfigDataType) -> Style {
319        let color = match data_type {
320            crate::ui::ConfigDataType::Integer { .. } => self.type_integer,
321            crate::ui::ConfigDataType::Float { .. } => self.type_float,
322            crate::ui::ConfigDataType::Boolean => self.type_boolean,
323            crate::ui::ConfigDataType::String => self.type_string,
324            crate::ui::ConfigDataType::Color => self.type_color,
325            crate::ui::ConfigDataType::Keyword { .. } => self.type_keyword,
326        };
327        Style::default().fg(color)
328    }

357impl std::fmt::Display for ColorScheme {
358    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
359        match self {
360            ColorScheme::Default => write!(f, "Default"),
361            ColorScheme::Gruvbox => write!(f, "Gruvbox"),
362            ColorScheme::Nord => write!(f, "Nord"),
363            ColorScheme::Catppuccin => write!(f, "Catppuccin"),
364            ColorScheme::Dracula => write!(f, "Dracula"),
365        }
366    }
367}
368
369impl FromStr for ColorScheme {
370    type Err = String;
371
372    fn from_str(s: &str) -> Result<Self, Self::Err> {
373        match s.to_lowercase().as_str() {
374            "default" => Ok(ColorScheme::Default),
375            "gruvbox" => Ok(ColorScheme::Gruvbox),
376            "nord" => Ok(ColorScheme::Nord),
377            "catppuccin" => Ok(ColorScheme::Catppuccin),
378            "dracula" => Ok(ColorScheme::Dracula),
379            _ => Err(format!("Unknown color scheme: {s}")),
380        }
381    }
382}
