55    /// Export structured configuration to specified format
56    pub fn export_structured_config(
57        &self,
58        config: &StructuredConfig,
59        name: &str,
60        format: ConfigFormat,
61    ) -> Result<PathBuf> {
62        // Ensure output directory exists
63        std::fs::create_dir_all(&self.output_dir)
64            .with_context(|| format!("Failed to create output directory: {:?}", self.output_dir))?;
65
66        let filename = format!("{}.{}", name, format.extension());
67        let output_path = self.output_dir.join(filename);
68
69        let content = match format {
70            ConfigFormat::HyprlandConf => self.to_hyprland_conf(config)?,
71            ConfigFormat::Json => self.to_json(config)?,
72            ConfigFormat::Toml => self.to_toml(config)?,
73            ConfigFormat::Yaml => self.to_yaml(config)?,
74            ConfigFormat::RHyprConfig => self.to_rhypr_config(config)?,
75            ConfigFormat::NixHomeManager => self.to_nix_home_manager(config)?,
76            ConfigFormat::NixSystem => self.to_nix_system(config)?,
77        };
78
79        std::fs::write(&output_path, content)
80            .with_context(|| format!("Failed to write config to: {:?}", output_path))?;
81
82        Ok(output_path)
83    }

217    /// Extract current configuration from hyprctl
218    async fn extract_current_config(
219        &self,
220        hyprctl: &crate::hyprctl::HyprCtl,
221    ) -> Result<StructuredConfig> {
222        let options = hyprctl.get_all_options().await?;
223        let binds = hyprctl.get_binds().await?;
224        let window_rules = hyprctl.get_window_rules().await?;
225        let layer_rules = hyprctl.get_layer_rules().await?;
226
227        let mut config = StructuredConfig::new("Current Configuration");
228
229        // Extract general settings
230        config.general.gaps_in = options.get("general:gaps_in").and_then(|v| v.parse().ok());
231        config.general.gaps_out = options.get("general:gaps_out").and_then(|v| v.parse().ok());
232        config.general.border_size = options
233            .get("general:border_size")
234            .and_then(|v| v.parse().ok());
235        config.general.col_active_border = options.get("general:col.active_border").cloned();
236        config.general.col_inactive_border = options.get("general:col.inactive_border").cloned();
237
238        // Extract keybinds
239        for bind in &binds {
240            config.keybinds.push(KeybindEntry {
241                bind_type: bind.bind_type.clone(),
242                modifiers: bind.modifiers.clone(),
243                key: bind.key.clone(),
244                dispatcher: bind.dispatcher.clone(),
245                args: bind.args.clone(),
246                description: None,
247            });
248        }
249
250        // Extract window rules
251        for rule in &window_rules {
252            let parts: Vec<&str> = rule.splitn(2, ',').collect();
253            if parts.len() == 2 {
254                config.window_rules.push(WindowRuleEntry {
255                    rule: parts[0].trim().to_string(),
256                    window_identifier: parts[1].trim().to_string(),
257                    description: None,
258                });
259            }
260        }
261
262        // Extract layer rules
263        for rule in &layer_rules {
264            let parts: Vec<&str> = rule.splitn(2, ',').collect();
265            if parts.len() == 2 {
266                config.layer_rules.push(LayerRuleEntry {
267                    rule: parts[0].trim().to_string(),
268                    layer: parts[1].trim().to_string(),
269                    description: None,
270                });
271            }
272        }
273
274        config.touch();
275        Ok(config)
276    }

280    fn to_hyprland_conf(&self, config: &StructuredConfig) -> Result<String> {
281        let mut conf = String::new();
282
283        // Header
284        conf.push_str(&format!(
285            "# {} - Generated by r-hyprconfig\n",
286            config.metadata.name
287        ));
288        conf.push_str(&format!(
289            "# Generated on: {}\n",
290            config.metadata.updated_at.format("%Y-%m-%d %H:%M:%S UTC")
291        ));
292        conf.push_str("\n");
293
294        // General settings
295        if let Some(gaps_in) = config.general.gaps_in {
296            conf.push_str(&format!("general:gaps_in = {}\n", gaps_in));
297        }
298        if let Some(gaps_out) = config.general.gaps_out {
299            conf.push_str(&format!("general:gaps_out = {}\n", gaps_out));
300        }
301        if let Some(border_size) = config.general.border_size {
302            conf.push_str(&format!("general:border_size = {}\n", border_size));
303        }
304        if let Some(ref color) = config.general.col_active_border {
305            conf.push_str(&format!("general:col.active_border = {}\n", color));
306        }
307        if let Some(ref color) = config.general.col_inactive_border {
308            conf.push_str(&format!("general:col.inactive_border = {}\n", color));
309        }
310
311        // Input settings
312        conf.push_str("\n# Input configuration\n");
313        if let Some(ref layout) = config.input.kb_layout {
314            conf.push_str(&format!("input:kb_layout = {}\n", layout));
315        }
316        if let Some(follow_mouse) = config.input.follow_mouse {
317            conf.push_str(&format!("input:follow_mouse = {}\n", follow_mouse));
318        }
319        if let Some(sensitivity) = config.input.sensitivity {
320            conf.push_str(&format!("input:sensitivity = {}\n", sensitivity));
321        }
322
323        // Decoration settings
324        conf.push_str("\n# Decoration\n");
325        if let Some(rounding) = config.decoration.rounding {
326            conf.push_str(&format!("decoration:rounding = {}\n", rounding));
327        }
328        if let Some(blur_enabled) = config.decoration.blur_enabled {
329            conf.push_str(&format!("decoration:blur:enabled = {}\n", blur_enabled));
330        }
331
332        // Animations
333        if let Some(enabled) = config.animations.enabled {
334            conf.push_str(&format!("\nanimations:enabled = {}\n", enabled));
335        }
336
337        // Keybinds
338        if !config.keybinds.is_empty() {
339            conf.push_str("\n# Keybinds\n");
340            for keybind in &config.keybinds {
341                let modifiers = if keybind.modifiers.is_empty() {
342                    String::new()
343                } else {
344                    format!("{}, ", keybind.modifiers.join(" + "))
345                };
346
347                let args = if let Some(ref args) = keybind.args {
348                    format!(", {}", args)
349                } else {
350                    String::new()
351                };
352
353                conf.push_str(&format!(
354                    "{} = {}{}, {}{}\n",
355                    keybind.bind_type, modifiers, keybind.key, keybind.dispatcher, args
356                ));
357            }
358        }
359
360        // Window rules
361        if !config.window_rules.is_empty() {
362            conf.push_str("\n# Window rules\n");
363            for rule in &config.window_rules {
364                conf.push_str(&format!(
365                    "windowrule = {}, {}\n",
366                    rule.rule, rule.window_identifier
367                ));
368            }
369        }
370
371        // Layer rules
372        if !config.layer_rules.is_empty() {
373            conf.push_str("\n# Layer rules\n");
374            for rule in &config.layer_rules {
375                conf.push_str(&format!("layerrule = {}, {}\n", rule.rule, rule.layer));
376            }
377        }
378
379        Ok(conf)
380    }

394    fn to_rhypr_config(&self, config: &StructuredConfig) -> Result<String> {
395        // Custom format with metadata and structured content
396        let mut content = String::new();
397
398        // Metadata header
399        content.push_str("# r-hyprconfig format\n");
400        content.push_str(&format!("# Name: {}\n", config.metadata.name));
401        if let Some(ref desc) = config.metadata.description {
402            content.push_str(&format!("# Description: {}\n", desc));
403        }
404        if let Some(ref author) = config.metadata.author {
405            content.push_str(&format!("# Author: {}\n", author));
406        }
407        content.push_str(&format!("# Version: {}\n", config.metadata.version));
408        content.push_str(&format!(
409            "# Created: {}\n",
410            config.metadata.created_at.format("%Y-%m-%d %H:%M:%S UTC")
411        ));
412        content.push_str(&format!(
413            "# Updated: {}\n",
414            config.metadata.updated_at.format("%Y-%m-%d %H:%M:%S UTC")
415        ));
416
417        if !config.metadata.tags.is_empty() {
418            content.push_str(&format!("# Tags: {}\n", config.metadata.tags.join(", ")));
419        }
420
421        content.push_str("\n");
422
423        // JSON content
424        content.push_str(&self.to_json(config)?);
425
426        Ok(content)
427    }

462    fn structured_to_hyprland_config(&self, config: &StructuredConfig) -> Result<HyprlandConfig> {
463        // Convert structured config back to HyprlandConfig for NixOS generator
464        let general = crate::hyprctl::GeneralConfig {
465            gaps_in: config.general.gaps_in.unwrap_or(5),
466            gaps_out: config.general.gaps_out.unwrap_or(20),
467            border_size: config.general.border_size.unwrap_or(2),
468            col_active_border: config
469                .general
470                .col_active_border
471                .clone()
472                .unwrap_or_else(|| "rgba(33ccffee)".to_string()),
473            col_inactive_border: config
474                .general
475                .col_inactive_border
476                .clone()
477                .unwrap_or_else(|| "rgba(595959aa)".to_string()),
478            resize_on_border: config.general.resize_on_border.unwrap_or(false),
479            extend_border_grab_area: config.general.extend_border_grab_area.unwrap_or(15),
480            hover_icon_on_border: config.general.hover_icon_on_border.unwrap_or(true),
481        };
482
483        let binds = config
484            .keybinds
485            .iter()
486            .map(|kb| {
487                format!(
488                    "{} {} -> {}",
489                    kb.modifiers.join(" + "),
490                    kb.key,
491                    kb.dispatcher
492                )
493            })
494            .collect();
495
496        let window_rules = config
497            .window_rules
498            .iter()
499            .map(|wr| format!("{}, {}", wr.rule, wr.window_identifier))
500            .collect();
501
502        let layer_rules = config
503            .layer_rules
504            .iter()
505            .map(|lr| format!("{}, {}", lr.rule, lr.layer))
506            .collect();
507
508        Ok(HyprlandConfig {
509            general,
510            input: Default::default(),
511            decoration: Default::default(),
512            animations: Default::default(),
513            gestures: Default::default(),
514            binds,
515            window_rules,
516            layer_rules,
517            misc: Default::default(),
518        })
519    }
